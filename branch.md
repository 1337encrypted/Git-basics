# List branches

`git branch`
`git branch --list`

# Create a new branch

`git checkout -b <branch-name>`

# Switch to another branch

`git checkout <branch>`

# show all local and remote branches

`git branch -a`

# Rename the main branch

`git branch -M main`

# Delete a branch

`git branch --delete <branchname>`
`git branch -d <branchname>`
`git branch -D <branchname>`
